:- write('[McLaren] MCL38 ready on the grid.'), send_m(semaphore, send_message(ready, mclaren)).
:- dynamic race_started/0.
:- dynamic race_over/0.
:- dynamic engine_failure_mclaren_fired/0.
:- dynamic push_lap_mclaren_fired/0.

start_raceE:>
    assert(race_started),
    retractall(push_lap_mclaren_fired),
    write('[McLaren] LAP 1 -- LIGHTS OUT! Norris launches off the line!'),
    messageA(pitwall, send_message(lap_done_mclaren, mclaren)).

lap_go_mclarenE:>
    if(\+ race_started, assert(race_started), true),
    retractall(push_lap_mclaren_fired),
    write('[McLaren] On the power! Norris attacking every sector.'),
    messageA(pitwall, send_message(lap_done_mclaren, mclaren)).

box_mclarenE:>
    write('[McLaren] BOX BOX BOX! McLaren dives into the pits.'),
    messageA(pitwall, send_message(pit_done_mclaren, mclaren)).

rain_warningE:>
    if(race_over, true,
        (write('[McLaren] RAIN WARNING. Norris switching to intermediate tyres.'), nl)).

safety_car_deployedE:>
    if(race_over, true,
        (write('[McLaren] Safety car deployed. Norris conserving tyres.'), nl)).

retire_mclarenE:>
    write('[McLaren] *** NORRIS PARKS THE CAR. McLaren is out of the race. ***').

race_endE:>
    assert(race_over).

engine_failure_mclaren :-
    race_started,
    \+ race_over,
    \+ engine_failure_mclaren_fired,
    random(0, 1000, R),
    R < 2.

engine_failure_mclarenI:>
    assert(engine_failure_mclaren_fired),
    write('[McLaren] *** ENGINE FAILURE -- Power loss! Norris slowing! ***'),
    send_m(pitwall, send_message(mclaren_engine_failure, mclaren)).

push_lap_mclaren :-
    race_started,
    \+ race_over,
    \+ push_lap_mclaren_fired,
    random(0, 100, R),
    R < 10.

push_lap_mclarenI:>
    assert(push_lap_mclaren_fired),
    write('[McLaren] PUSH LAP! Norris going flat out -- setting fastest sectors!'),
    send_m(pitwall, send_message(mclaren_push_lap, mclaren)).
