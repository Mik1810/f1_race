:- write('[Red Bull] RB20 ready on the grid.'), send_m(semaphore, send_message(ready, redbull)).
:- dynamic race_started/0.
:- dynamic race_over/0.
:- dynamic engine_failure_redbull_fired/0.
:- dynamic push_lap_redbull_fired/0.

start_raceE:>
    assert(race_started),
    retractall(push_lap_redbull_fired),
    write('[Red Bull] LAP 1 -- LIGHTS OUT! Verstappen launches off the line!'),
    messageA(pitwall, send_message(lap_done_redbull, redbull)).

lap_go_redbullE:>
    if(\+ race_started, assert(race_started), true),
    retractall(push_lap_redbull_fired),
    write('[Red Bull] On the power! Verstappen attacking every sector.'),
    messageA(pitwall, send_message(lap_done_redbull, redbull)).

box_redbullE:>
    write('[Red Bull] BOX BOX BOX! Red Bull dives into the pits.'),
    messageA(pitwall, send_message(pit_done_redbull, redbull)).

rain_warningE:>
    if(race_over, true,
        (write('[Red Bull] RAIN WARNING. Verstappen switching to intermediate tyres.'), nl)).

safety_car_deployedE:>
    if(race_over, true,
        (write('[Red Bull] Safety car deployed. Verstappen conserving tyres.'), nl)).

retire_redbullE:>
    write('[Red Bull] *** VERSTAPPEN PARKS THE CAR. Red Bull is out of the race. ***').

green_flagE:>
    if(race_over, true,
        (write('[Red Bull] GREEN FLAG! Verstappen pushing flat out!'), nl)).

race_endE:>
    assert(race_over).

engine_failure_redbull :-
    race_started,
    \+ race_over,
    \+ engine_failure_redbull_fired,
    random(0, 1000, R),
    R < 2.

engine_failure_redbullI:>
    assert(engine_failure_redbull_fired),
    write('[Red Bull] *** ENGINE FAILURE -- Power loss! Verstappen slowing! ***'),
    send_m(pitwall, send_message(redbull_engine_failure, redbull)).

push_lap_redbull :-
    race_started,
    \+ race_over,
    \+ push_lap_redbull_fired,
    random(0, 100, R),
    R < 10.

push_lap_redbullI:>
    assert(push_lap_redbull_fired),
    write('[Red Bull] PUSH LAP! Verstappen going flat out -- setting fastest sectors!'),
    send_m(pitwall, send_message(redbull_push_lap, redbull)).
